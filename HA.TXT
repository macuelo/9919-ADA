import java.util.*;

Map<String, List<Integer>> invertedIndex = new HashMap<>();

// Ejemplo: indexar documentos
void addToIndex(String word, int docId) {
    invertedIndex.computeIfAbsent(word, k -> new ArrayList<>()).add(docId);
}

// BÃºsqueda
List<Integer> getDocuments(String word) {
    return invertedIndex.getOrDefault(word, Collections.emptyList());
}
